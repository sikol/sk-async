cmake_minimum_required(VERSION 3.12)

project(sk-cio VERSION 1.0.0)

enable_testing()

option(SK_CIO_USE_CLANG_TIDY "Run clang-tidy during build")

find_program(SK_CIO_CLANG_TIDY_EXE NAMES "clang-tidy" DOC "Path to clang-tidy executable" )

if(NOT SK_CIO_CLANG_TIDY_EXE)
    message(STATUS "clang-tidy not found.")
else()
    message(STATUS "clang-tidy found: ${SK_CIO_CLANG_TIDY_EXE}")
    if(SK_CIO_USE_CLANG_TIDY)
        set(SK_CIO_ENABLE_TIDY ON)
        set(SK_CLANG_TIDY_COMMAND "${SK_CIO_CLANG_TIDY_EXE};--system-headers")
    endif()
endif()

add_subdirectory(thirdparty)
add_subdirectory(tests)
add_subdirectory(src)
add_subdirectory(sample)
